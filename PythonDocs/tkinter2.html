Advanced Tkinter Layout & Geometry Managers

Beyond simple pack() usage, Tkinter offers powerful layout control with pack, grid, and place. You can also combine frames strategically to build complex responsive interfaces.

Geometry Manager Overview
  • pack(): Great for stacking widgets vertically or horizontally; uses side, fill, expand.
  • grid(): Table-like rows/columns; supports weight-based resizing.
  • place(): Absolute or relative positioning (x/y, relx/rely); least flexible for responsive UIs.

Advanced pack(): Vertical vs Horizontal

Use nested frames and the side, fill, and expand options to control distribution.

# Advanced vertical + horizontal stacking with pack() import tkinter as tk root = tk.Tk() root.title("pack() Advanced") root.geometry("420x260") # Top bar (horizontal) top_bar = tk.Frame(root, bg="#222", height=40) top_bar.pack(side="top", fill="x") for txt in ["File", "Edit", "View"]: tk.Label(top_bar, text=txt, fg="white", bg="#222", padx=12).pack(side="left") # Main area split left nav / right content main = tk.Frame(root) main.pack(fill="both", expand=True) nav = tk.Frame(main, bg="#333", width=120) nav.pack(side="left", fill="y") content = tk.Frame(main, bg="#111") content.pack(side="left", fill="both", expand=True) # Add some vertical buttons in nav for i in range(1, 6): tk.Button(nav, text=f"Option {i}", relief="flat").pack(fill="x", pady=2, padx=6) # Content area demonstrates vertical expansion header = tk.Label(content, text="Dashboard", font=("Segoe UI", 14, "bold"), bg="#111", fg="white") header.pack(anchor="w", padx=10, pady=(10,6)) cards = tk.Frame(content, bg="#111") cards.pack(fill="both", expand=True) # Create row of 3 pseudo cards horizontally inside cards frame for label in ["Users", "Sales", "Errors"]: box = tk.Frame(cards, bg="#222", bd=1, relief="ridge", padx=10, pady=8) box.pack(side="left", fill="both", expand=True, padx=6, pady=4) tk.Label(box, text=label, fg="white", bg="#222", font=("Segoe UI", 12, "bold")).pack(anchor="w") tk.Label(box, text="Value: 0", fg="white", bg="#222").pack(anchor="w", pady=4) root.mainloop()

Responsive grid() with Weight

Assign a weight to rows and columns so space distributes proportionally during window resize.

# Responsive grid() layout using row/column weights import tkinter as tk root = tk.Tk() root.title("grid() Weights") root.geometry("420x240") for c in range(3): root.columnconfigure(c, weight=1) for r in range(3): root.rowconfigure(r, weight=1) colors = ["#394", "#349", "#933", "#884", "#268", "#666", "#055", "#550", "#333"] # Fill the 3x3 grid idx = 0 for r in range(0, 3): for c in range(0, 3): frame = tk.Frame(root, bg=colors[idx], bd=1) frame.grid(row=r, column=c, sticky="nsew", padx=2, pady=2) tk.Label(frame, text=f"{r},{c}", bg=colors[idx], fg="white").pack(expand=True) idx += 1 root.mainloop()

Mixing pack() and grid() Safely

You can use pack in one container (frame) and grid in another. Just never mix them within the same container.

# pack() in root, grid() inside a form frame import tkinter as tk root = tk.Tk() root.title("Mix pack & grid") root.geometry("360x200") banner = tk.Label(root, text="User Login", font=("Segoe UI", 14, "bold")) banner.pack(pady=8) form = tk.Frame(root, padx=10, pady=10) form.pack(fill="x") # grid inside form for i in range(0, 2): form.columnconfigure(i, weight=1) labels = ["Username:", "Password:"] entries = [] for r, text in enumerate(labels): tk.Label(form, text=text).grid(row=r, column=0, sticky="e", pady=4) e = tk.Entry(form, show="*") if "Password" in text else tk.Entry(form) e.grid(row=r, column=1, sticky="ew", pady=4) entries.append(e) status = tk.Label(root, text="Ready", fg="gray") status.pack(side="bottom", fill="x") root.mainloop()

# place() absolute + relative demo import tkinter as tk root = tk.Tk() root.title("place() Demo") root.geometry("360x220") canvas = tk.Frame(root, bg="#222") canvas.pack(fill="both", expand=True) # Absolute box1 = tk.Label(canvas, text="(20,20)", bg="#e91e63", fg="white") box1.place(x=20, y=20) # Relative center box2 = tk.Label(canvas, text="center", bg="#2196f3", fg="white") box2.place(relx=0.5, rely=0.5, anchor="center") # Relative bottom right box3 = tk.Label(canvas, text="br", bg="#4caf50", fg="white") box3.place(relx=0.95, rely=0.95, anchor="se") root.mainloop()

Scrollable Frame Pattern

Tkinter lacks a native scrollable frame, but you can create one with Canvas + Frame + scrollbar.

# Reusable scrollable frame pattern import tkinter as tk class ScrollFrame(tk.Frame): def __init__(self, master, *args, **kwargs): super().__init__(master, *args, **kwargs) canvas = tk.Canvas(self, highlightthickness=0) vsb = tk.Scrollbar(self, orient="vertical", command=canvas.yview) self.inner = tk.Frame(canvas) self.inner.bind("", lambda e: canvas.configure(scrollregion=canvas.bbox("all"))) canvas.create_window((0,0), window=self.inner, anchor="nw") canvas.configure(yscrollcommand=vsb.set) canvas.pack(side="left", fill="both", expand=True) vsb.pack(side="right", fill="y") root = tk.Tk() root.title("Scrollable Frame") root.geometry("300x260") scroll = ScrollFrame(root) scroll.pack(fill="both", expand=True) for i in range(1, 51): tk.Label(scroll.inner, text=f"Item {i}").pack(anchor="w", padx=8, pady=2) root.mainloop()

Dynamic Column / Row Spanning

Use columnspan and rowspan to merge cells for headers or sidebars.

# columnspan / rowspan usage import tkinter as tk root = tk.Tk() root.title("grid() span") root.geometry("400x260") for c in range(4): root.columnconfigure(c, weight=1) for r in range(6): root.rowconfigure(r, weight=1) header = tk.Label(root, text="Report Overview", bg="#222", fg="white") header.grid(row=0, column=0, columnspan=4, sticky="nsew") sidebar = tk.Label(root, text="Filters", bg="#394", fg="white") sidebar.grid(row=1, column=0, rowspan=5, sticky="nsew") for r in range(1, 6): for c in range(1, 4): tk.Label(root, text=f"{r},{c}", bg="#333", fg="white").grid(row=r, column=c, sticky="nsew", padx=1, pady=1) root.mainloop()
